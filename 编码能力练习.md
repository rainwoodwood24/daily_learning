```c

{
```

```c
#include<stdio.h>
#define MAX_N 1000

{
    int sum1=(3+999)*(999/3)/2;

```

>   [!IMPORTANT]

**定理：**a和b两个整数的最大公约数等于b与a%b的最大公约数

```c
#include<stdio.h>
int gcd(int a,int b){
    if(b==0) return a;
    return gcd(b,a%b);

    printf("%d\n",gcd(a,b));
    return 0;
}
```

## 线性筛：比素数筛高效

素数筛：一个合数可能被多次标记

**线性筛：时间复杂度：O（n)     空间复杂度：O（n)**

算法：利用M标记整数N，其中M是除N外最大的因子，N=M*p;

eg:若N=30，则算法中的M、p分别为15,2

若M=25，则算法中的N都有哪些？ 50,75,125

找到规律：M%p==0，则M*p=N(最大)

```c
int prime[MAX_N+1]={0};
void is_prime(){
    for(int i=2;i<=MAX_N;i++){
        if(!prime[i]) prime[++prime[0]]=i;
        for(int j=1;j<=prime[0];j++){
            if(prime[j]*i>MAX_N) break;
            prime[prime[j]*i]=1;
            if(i%prime[j]==0) break;
        }
    }
    return ;
}
```

### 5、EP37:可截素数

如何去掉n的最低位，如何去掉n的最高位

```c
#include<stdio.h>
#include<math.h>
#define MAX_N 2000000

int prime[MAX_N + 3] = {0};
int is_prime[MAX_N + 2] = {0, 1};
void init_prime() {
	for (int i = 2; i <= MAX_N; i++) {
		if (!is_prime[i]) prime[++prime[0]] = i;
		for (int j = 1; j <= prime[0]; j++) {
			if (prime[j]*i > MAX_N) break;
			is_prime[prime[j]*i] = 1;
			if (i % prime[j] == 0) break;
		}
	}
	return ;
}

int is_value(int n) {
	int h = pow(10, floor(log10(n))), temp = n;
	//left
	while (n) {
		if (is_prime[n]) return 0;
		n %= h;
		h /= 10;
	}
	//right
	while (temp) {
		if (is_prime[temp]) return 0;
		temp = temp / 10;
	}
	return 1;
}

int main() {
	init_prime();
	int sum = 0, flag = 11;
	for (int i = 5; i <= prime[i] && flag; i++) {
		if (!is_value(prime[i])) continue;
		sum += prime[i];
		flag--;
	}
	printf("%d\n", sum);
```
